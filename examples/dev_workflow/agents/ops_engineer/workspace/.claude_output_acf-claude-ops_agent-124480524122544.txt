```bash
#!/bin/bash

# ACF v2 Deployment Script
# Usage: ./deploy.sh [environment] [version]
# Example: ./deploy.sh production v2.0.0

set -euo pipefail

# Configuration
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly ENVIRONMENT="${1:-staging}"
readonly VERSION="${2:-latest}"
readonly APP_NAME="acf-v2"
readonly DEPLOY_USER="deploy"
readonly BACKUP_DIR="/opt/backups/${APP_NAME}"
readonly DEPLOY_DIR="/opt/apps/${APP_NAME}"
readonly LOG_DIR="/var/log/${APP_NAME}"
readonly HEALTH_CHECK_URL="http://localhost:8000/health"
readonly HEALTH_CHECK_TIMEOUT=30
readonly MAX_RETRIES=5

# Colors for output
readonly RED='\033[0;31m'
readonly GREEN='\033[0;32m'
readonly YELLOW='\033[1;33m'
readonly NC='\033[0m' # No Color

# Logging functions
log_info() {
    echo -e "${GREEN}[INFO]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1"
}

# Pre-deployment checks
pre_deploy_checks() {
    log_info "Running pre-deployment checks..."

    # Check if running as deploy user
    if [[ "$(whoami)" != "${DEPLOY_USER}" ]]; then
        log_error "Must run as ${DEPLOY_USER} user"
        exit 1
    fi

    # Check Python version
    if ! python3 --version | grep -q "Python 3.12"; then
        log_error "Python 3.12 is required"
        exit 1
    fi

    # Check disk space (min 1GB)
    local available_space
    available_space=$(df "${DEPLOY_DIR}" | awk 'NR==2 {print $4}')
    if [[ "${available_space}" -lt 1048576 ]]; then
        log_error "Insufficient disk space. Required: 1GB"
        exit 1
    fi

    # Check if virtual environment exists
    if [[ ! -d "${DEPLOY_DIR}/venv" ]]; then
        log_warn "Virtual environment not found, will create one"
    fi

    log_info "Pre-deployment checks passed"
}

# Create backup
create_backup() {
    log_info "Creating backup..."

    local backup_timestamp
    backup_timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_path="${BACKUP_DIR}/backup_${backup_timestamp}"

    mkdir -p "${backup_path}"

    if [[ -d "${DEPLOY_DIR}" ]]; then
        cp -r "${DEPLOY_DIR}" "${backup_path}/"
        log_info "Backup created at ${backup_path}"
    else
        log_warn "No existing deployment to backup"
    fi

    # Keep only last 5 backups
    ls -t "${BACKUP_DIR}" | tail -n +6 | xargs -r rm -rf

    echo "${backup_path}" > /tmp/last_backup_path
}

# Deploy application
deploy() {
    log_info "Starting deployment of ${APP_NAME} ${VERSION} to ${ENVIRONMENT}..."

    # Create directories
    mkdir -p "${DEPLOY_DIR}" "${LOG_DIR}" "${BACKUP_DIR}"

    # Clone/pull latest code
    if [[ -d "${DEPLOY_DIR}/.git" ]]; then
        cd "${DEPLOY_DIR}"
        git fetch origin
        git checkout "${VERSION}"
        git pull origin "${VERSION}"
    else
        git clone --branch "${VERSION}" https://github.com/org/acf-v2.git "${DEPLOY_DIR}"
        cd "${DEPLOY_DIR}"
    fi

    # Setup virtual environment
    if [[ ! -d "venv" ]]; then
        python3 -m venv venv
    fi
    source venv/bin/activate

    # Install dependencies
    pip install --upgrade pip
    pip install -e .
    pip install -r requirements.txt 2>/dev/null || log_warn "No requirements.txt found"

    # Run database migrations (if applicable)
    # alembic upgrade head || log_warn "No migrations to run"

    # Run tests before starting
    log_info "Running tests..."
    if ! python -m pytest tests/ -q --tb=short; then
        log_error "Tests failed! Aborting deployment."
        exit 1
    fi

    log_info "Deployment completed successfully"
}

# Health check
health_check() {
    log_info "Running health checks..."

    local retries=0

    # Wait for service to start
    sleep 5

    while [[ ${retries} -lt ${MAX_RETRIES} ]]; do
        if curl -sf "${HEALTH_CHECK_URL}" > /dev/null 2>&1; then
            log_info "Health check passed"
            return 0
        fi

        retries=$((retries + 1))
        log_warn "Health check failed, retry ${retries}/${MAX_RETRIES}..."
        sleep 5
    done

    log_error "Health check failed after ${MAX_RETRIES} attempts"
    return 1
}

# Start application
start_application() {
    log_info "Starting application..."

    cd "${DEPLOY_DIR}"
    source venv/bin/activate

    # Stop existing process
    if pgrep -f "acf" > /dev/null; then
        log_info "Stopping existing application..."
        pkill -f "acf" || true
        sleep 2
    fi

    # Start application (adjust based on actual entry point)
    nohup python -m acf.server > "${LOG_DIR}/app.log" 2>&1 &

    # Save PID
    echo $! > /tmp/acf.pid

    log_info "Application started with PID $(cat /tmp/acf.pid)"
}

# Rollback function
rollback() {
    log_error "Deployment failed! Initiating rollback..."

    local last_backup
    last_backup=$(cat /tmp/last_backup_path 2>/dev/null || echo "")

    if [[ -z "${last_backup}" ]] || [[ ! -d "${last_backup}" ]]; then
        log_error "No backup found for rollback!"
        exit 1
    fi

    # Stop current application
    if [[ -f /tmp/acf.pid ]]; then
        kill "$(cat /tmp/acf.pid)" 2>/dev/null || true
        rm -f /tmp/acf.pid
    fi

    # Restore from backup
    rm -rf "${DEPLOY_DIR}"
    cp -r "${last_backup}/$(basename "${DEPLOY_DIR}")" "${DEPLOY_DIR}"

    # Restart application
    cd "${DEPLOY_DIR}"
    source venv/bin/activate
    nohup python -m acf.server > "${LOG_DIR}/app.log" 2>&1 &
    echo $! > /tmp/acf.pid

    log_info "Rollback completed. Application restored from ${last_backup}"
}

# Cleanup function
cleanup() {
    log_info "Cleaning up..."
    # Remove temporary files
    rm -f /tmp/acf_deploy_*.tmp
}

# Main deployment flow
main() {
    trap cleanup EXIT

    log_info "=========================================="
    log_info "Deployment started: ${APP_NAME} ${VERSION}"
    log_info "Environment: ${ENVIRONMENT}"
    log_info "=========================================="

    # Execute deployment steps
    if ! pre_deploy_checks; then
        log_error "Pre-deployment checks failed"
        exit 1
    fi

    create_backup

    if ! deploy; then
        rollback
        exit 1
    fi

    start_application

    if ! health_check; then
        rollback
        exit 1
    fi

    log_info "=========================================="
    log_info "Deployment successful!"
    log_info "Application: ${APP_NAME}"
    log_info "Version: ${VERSION}"
    log_info "Environment: ${ENVIRONMENT}"
    log_info "=========================================="
}

# Handle command line arguments
case "${1:-}" in
    --rollback)
        rollback
        exit 0
        ;;
    --health-check)
        health_check
        exit 0
        ;;
    --stop)
        if [[ -f /tmp/acf.pid ]]; then
            kill "$(cat /tmp/acf.pid)" 2>/dev/null && log_info "Application stopped"
            rm -f /tmp/acf.pid
        else
            log_warn "No PID file found"
        fi
        exit 0
        ;;
    --status)
        if [[ -f /tmp/acf.pid ]] && kill -0 "$(cat /tmp/acf.pid)" 2>/dev/null; then
            log_info "Application is running (PID: $(cat /tmp/acf.pid))"
        else
            log_warn "Application is not running"
        fi
        exit 0
        ;;
    --help|-h)
        echo "Usage: $0 [environment] [version]"
        echo "       $0 --rollback"
        echo "       $0 --health-check"
        echo "       $0 --stop"
        echo "       $0 --status"
        exit 0
        ;;
    *)
        main
        ;;
esac
```
